## 1. 项目概述

### 1.1 项目定位
货运航空公司的智能化机组排班与资质管理系统，实现航班自动排班、合规实时校验、资质到期预警。

### 1.2 核心目标
- 提高排班效率，减少人工干预
- 确保符合民航局 AC/OPS/013R00 标准
- 降低违规风险，保障飞行安全

CREATE TABLE crew_base (
    crew_id BIGINT PRIMARY KEY AUTO_INCREMENT,

    crew_type VARCHAR(20) NOT NULL COMMENT '类型：captain/first_officer/standby',
    qualification_level VARCHAR(20) COMMENT '资质等级：leader/route/standby',

    status VARCHAR(20) DEFAULT 'onduty' COMMENT '状态：onduty/standby/rest/training',

    flight_hours_28d_limit DECIMAL(10,2) DEFAULT 100 COMMENT '28天飞行时长限制',
    seven_day_duty DECIMAL(10,2) DEFAULT 0 COMMENT '7天执勤时长',
    seven_day_duty_limit DECIMAL(10,2) DEFAULT 55 COMMENT '7天执勤时长限制',
    simulator_hours DECIMAL(10,2) DEFAULT 0 COMMENT '模拟机时长',
    training_hours DECIMAL(10,2) DEFAULT 0 COMMENT '地面培训时长',
    duty_hours DECIMAL(10,2) DEFAULT 0 COMMENT '值班时长',
    last_rest_hours DECIMAL(10,2) DEFAULT 0 COMMENT '最近休息时长',
    license_expiry DATE COMMENT '证照到期日期',
    medical_expiry DATE COMMENT '体检到期日期',
    compliance_status VARCHAR(20) DEFAULT 'normal' COMMENT '合规状态',
    del_flag CHAR(1) DEFAULT '0' COMMENT '删除标志',
    create_by VARCHAR(64) COMMENT '创建者',

    update_by VARCHAR(64) COMMENT '更新者',

CREATE TABLE crew_medical (
    medical_id BIGINT PRIMARY KEY AUTO_INCREMENT,

    create_time DATETIME DEFAULT CURRENT_TIMESTAMP,

-- 机组培训记录表

    training_id BIGINT PRIMARY KEY AUTO_INCREMENT,

    training_type VARCHAR(20) COMMENT '训练类型：SIM/Ground School/Flight',

    training_location VARCHAR(200) COMMENT '训练地点',
    instructor VARCHAR(100) COMMENT '教员',
    result VARCHAR(20) COMMENT '结果：通过/未通过',
    regulation_reference VARCHAR(200) COMMENT '法规依据',
    remark VARCHAR(500) COMMENT '备注',
    status VARCHAR(20) DEFAULT 'valid' COMMENT '状态',
